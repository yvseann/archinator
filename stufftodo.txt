# ok we do the username now
while true; do
    read -p "Enter the username you want (e.g. yvseann): " USER_NAME

    # no empty input pls
    if [[ -z "$USER_NAME" ]]; then
        echo "Username cannot be empty."
        continue
    fi

    # is it valid
    if [[ "$USER_NAME" =~ ^[a-z_][a-z0-9_-]*$ ]]; then
        echo "Username accepted: $USER_NAME"
        break
    else
        echo "Invalid username. Usernames must:"
        echo "  - start with a lowercase letter or underscore"
        echo "  - contain only lowercase letters, numbers, underscores, or hyphens"
    fi
done
